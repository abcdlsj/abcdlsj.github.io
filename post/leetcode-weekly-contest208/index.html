5524. 经营摩天轮的最大利润

写得很仓促。。。

class Solution {
public:
    int minOperationsMaxProfit(vector<int>& c, int b, int r) {
        int time = 0, ans = INT_MIN, cur = 0, sum = 0, anstime = 0;
        
        for (int i = 0; i < c.size(); i++) {
            sum += c[i];
            time++;
            if (sum >= 4) {
                cur += 4;
                sum -= 4;
            } else {
                cur += sum;
                sum = 0;
            }
            if (cur * b - r * time > ans) {
                ans = cur * b - r * time;
                anstime = time;
            }
        }
        while (sum) {
            time++;
            if (sum >= 4) {
                cur += 4;
                sum -= 4;
            } else {
                cur += sum;
                sum = 0;
            }
            if (cur * b - r * time > ans) {
                ans = cur * b - r * time;
                anstime = time;
            }
        }
        
        return ans <= 0 ? -1 : anstime;
    }
};

5525. 皇位继承顺序

写得很仓促。。。

class Node {
public:
    bool isalive;
    string name;
    vector<Node*> childs;
    Node(string _name) : name(_name), isalive(true) {};
};
class ThroneInheritance {
public:
    Node* fam;
    unordered_map<string, Node*> map;
    string king;
    ThroneInheritance(string kingName) {
        fam = new Node(kingName);
        map[kingName] = fam;
        king = kingName;
    }
    
    void birth(string parentName, string childName) {
        Node* c = new Node(childName);
        map[parentName]->childs.push_back(c);
        map[childName] = map[parentName]->childs.back();
    }
    
    void death(string name) {
        map[name]->isalive = false;
    }
    vector<string> res;
    vector<string> getInheritanceOrder() {
        res.clear();
        midorder(map[king]);
        return res;
    }
    void midorder(Node* root) {
        if (root == nullptr) return ;
        if (map[root->name]->isalive) {
            res.push_back(root->name);
        }
        for (auto& c : root->childs) midorder(c); 
    }
};
